list_datasets_csv=("data_quiz/wdi_co2_2005.csv","data_quiz/wdi_co2_2010.csv","data_quiz/wdi_co2_2015.csv")
index=0
#use year vector from before plus the index to include the year in the title
for name in list_datasets_csv:
    dataset=pd.read_csv(name)
    co2_gasfuel_kt = dataset["co2_gasfuel_kt"] 
    co2_liquidfuel_kt = dataset["co2_liquidfuel_kt"]
    totalCO2emissions=co2_gasfuel_kt+co2_liquidfuel_kt
