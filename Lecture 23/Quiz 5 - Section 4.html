  • Generate a line plot with the "growth_cb" series, label the axes and the title

bitcoin["bool_cond"] = (bitcoin["growth_cb"] > 15) | (bitcoin["growth_cb"] < -15)

mat.plot("date", "growth_cb", data = bitcoin)
mat.xlabel("Time")
mat.ylabel("Daily Percentage Change of Bitcoin Exchange Rate")
mat.title("Daily Percentage Change of Bitcoin Exchange Rate over Time")

vec_y = bitcoin["growth_cb"]
mat.fill_between(x= bitcoin["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = bitcoin["bool_cond"],
                 alpha = 0.2,color = "red")

mat.show()

# Write your own code
month_config = pd.Grouper(key='date', freq='m')
monthlydata = (bitcoin
               .groupby(month_config) 
               .agg(Bitcoin_exchange_rate_standard_deviation = ("cb_to_usd","std"))
               .reset_index())
